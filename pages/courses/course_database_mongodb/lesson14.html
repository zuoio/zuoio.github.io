14.2.6 设计决策分析：为什么评论使用独立集合

在文档模型设计中，评论数据的存放方式是一个典型的决策点。我们选择 独立集合（Reference） 而非 内嵌文档（Embed） ，原因如下：

考量维度 内嵌方案 独立集合方案（本项目选择）
文档大小 热门文章可能有成百上千条评论，单文档会逼近 16MB 限制 每条评论独立存储，无大小风险
查询灵活性 难以单独对评论做分页、排序、聚合 可独立查询、分页、按时间排序
写入冲突 多用户同时评论会导致对同一文档的写竞争 各评论独立写入，无竞争
更新性能 修改深层嵌套数据需要位置操作符，性能较差 直接通过 _id 更新，简单高效
独立生命周期 评论的增删不应影响文章文档的版本和更新时间 评论与文章解耦，互不影响
级联删除 删除文章时需手动清理内嵌数据（但本身就在文档内） 需要事务保证删除文章时同步删评论

结论：当子数据量可能增长到不确定规模、且需要独立查询时，应选择独立集合 + 引用关联。

14.2.7 索引设计方案

// users 集合
db.users.createIndex({ "username": 1 }, { unique: true })
db.users.createIndex({ "email": 1 }, { unique: true })

// articles 集合
db.articles.createIndex({ "author_id": 1 })
db.articles.createIndex({ "category_id": 1, "status": 1, "created_at": -1 })
db.articles.createIndex({ "tags": 1 })
db.articles.createIndex({ "status": 1, "created_at": -1 })
db.articles.createIndex(
  { "title": "text", "summary": "text", "content": "text" },
  { weights: { "title": 10, "summary": 5, "content": 1 }, default_language: "none" }
)

// categories 集合
db.categories.createIndex({ "name": 1 }, { unique: true })
db.categories.createIndex({ "slug": 1 }, { unique: true })
db.categories.createIndex({ "parent_id": 1 })

// comments 集合
db.comments.createIndex({ "article_id": 1, "created_at": -1 })
db.comments.createIndex({ "user_id": 1 })

// verification_codes 集合
db.verification_codes.createIndex({ "email": 1 }, { unique: true })
db.verification_codes.createIndex({ "created_at": 1 }, { expireAfterSeconds: 300 })

索引设计思路：遵循 ESR 原则（Equality → Sort → Range），将等值查询字段放在前面，排序字段居中，范围查询字段放在最后。例如文章列表查询通常先按 category_id 和 status 精确过滤，再按 created_at 倒序排列，因此复合索引设计为 {category_id: 1, status: 1, created_at: -1}。

package com.cms.common;

import lombok.extern.slf4j.Slf4j;
import org.springframework.dao.DuplicateKeyException;
import org.springframework.http.HttpStatus;
import org.springframework.validation.FieldError;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestControllerAdvice;

import java.util.stream.Collectors;

@Slf4j
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(IllegalArgumentException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Result<Void> handleIllegalArgument(IllegalArgumentException e) {
        log.warn("参数错误: {}", e.getMessage());
        return Result.badRequest(e.getMessage());
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Result<Void> handleValidation(MethodArgumentNotValidException e) {
        String msg = e.getBindingResult().getFieldErrors().stream()
                .map(FieldError::getDefaultMessage)
                .collect(Collectors.joining("; "));
        log.warn("参数校验失败: {}", msg);
        return Result.badRequest(msg);
    }

    @ExceptionHandler(DuplicateKeyException.class)
    @ResponseStatus(HttpStatus.CONFLICT)
    public Result<Void> handleDuplicateKey(DuplicateKeyException e) {
        log.warn("唯一键冲突: {}", e.getMessage());
        return Result.error(409, "数据已存在，请检查唯一字段");
    }

    @ExceptionHandler(RuntimeException.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public Result<Void> handleRuntime(RuntimeException e) {
        log.error("运行时异常", e);
        return Result.error("服务器内部错误: " + e.getMessage());
    }

    @ExceptionHandler(Exception.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public Result<Void> handleException(Exception e) {
        log.error("未知异常", e);
        return Result.error("服务器内部错误");
    }
}

package com.cms.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import javax.crypto.SecretKey;
import java.nio.charset.StandardCharsets;
import java.util.Date;
import java.util.List;

@Component
public class JwtUtil {

    private final SecretKey key;
    private final long expiration;

    public JwtUtil(@Value("${jwt.secret}") String secret,
                   @Value("${jwt.expiration}") long expiration) {
        this.key = Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
        this.expiration = expiration;
    }

    public String generateToken(String userId, String username, List<String> roles) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + expiration);

        return Jwts.builder()
                .subject(userId)
                .claim("username", username)
                .claim("roles", roles)
                .issuedAt(now)
                .expiration(expiryDate)
                .signWith(key)
                .compact();
    }

    public Claims parseToken(String token) {
        return Jwts.parser()
                .verifyWith(key)
                .build()
                .parseSignedClaims(token)
                .getPayload();
    }

    public String getUserIdFromToken(String token) {
        return parseToken(token).getSubject();
    }

    public boolean validateToken(String token) {
        try {
            parseToken(token);
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}

@Service
@RequiredArgsConstructor
public class UserService {

    private final UserRepository userRepository;
    private final JwtUtil jwtUtil;
    private final BCryptPasswordEncoder passwordEncoder = new BCryptPasswordEncoder();

    public User register(RegisterRequest request) {
        if (userRepository.existsByUsername(request.getUsername())) {
            throw new IllegalArgumentException("用户名已存在");
        }
        if (userRepository.existsByEmail(request.getEmail())) {
            throw new IllegalArgumentException("邮箱已被注册");
        }

        User user = new User();
        user.setUsername(request.getUsername());
        user.setPassword(passwordEncoder.encode(request.getPassword()));
        user.setNickname(request.getNickname());
        user.setEmail(request.getEmail());
        user.setRoles(List.of("ROLE_USER"));
        user.setStatus(1);

        return userRepository.save(user);
    }

    public Map<String, Object> login(LoginRequest request) {
        User user = userRepository.findByUsername(request.getUsername())
                .orElseThrow(() -> new IllegalArgumentException("用户名或密码错误"));

        if (!passwordEncoder.matches(request.getPassword(), user.getPassword())) {
            throw new IllegalArgumentException("用户名或密码错误");
        }

        if (user.getStatus() == 0) {
            throw new IllegalArgumentException("账号已被禁用");
        }

        String token = jwtUtil.generateToken(user.getId(), user.getUsername(), user.getRoles());

        Map<String, Object> result = new HashMap<>();
        result.put("token", token);
        result.put("userId", user.getId());
        result.put("username", user.getUsername());
        result.put("nickname", user.getNickname());
        result.put("roles", user.getRoles());
        return result;
    }

    public User getUserById(String id) {
        User user = userRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("用户不存在"));
        user.setPassword(null);
        return user;
    }

package com.cms.service;

import com.cms.dto.ArticleQuery;

import com.cms.repository.ArticleRepository;
import com.cms.repository.CommentRepository;
import com.cms.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.*;
import org.springframework.data.mongodb.core.MongoTemplate;
import org.springframework.data.mongodb.core.query.Criteria;
import org.springframework.data.mongodb.core.query.Query;
import org.springframework.data.mongodb.core.query.Update;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.util.StringUtils;

import java.util.ArrayList;
import java.util.List;

@Service
@RequiredArgsConstructor
public class ArticleService {

    private final ArticleRepository articleRepository;
    private final CommentRepository commentRepository;
    private final UserRepository userRepository;
    private final MongoTemplate mongoTemplate;

    public Article publish(ArticleRequest request, String userId) {
        User author = userRepository.findById(userId)
                .orElseThrow(() -> new IllegalArgumentException("用户不存在"));

        Article article = new Article();
        article.setTitle(request.getTitle());
        article.setSummary(request.getSummary());
        article.setContent(request.getContent());
        article.setAuthorId(userId);
        article.setAuthorName(author.getNickname());
        article.setCategoryId(request.getCategoryId());
        article.setCategoryName(request.getCategoryName());
        article.setTags(request.getTags());
        article.setCoverImage(request.getCoverImage());
        article.setStatus(request.getStatus() != null ? request.getStatus() : "DRAFT");
        article.setViewCount(0L);
        article.setCommentCount(0);
        article.setIsTop(false);

        return articleRepository.save(article);
    }

    public Article update(String id, ArticleRequest request, String userId) {
        Article article = articleRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("文章不存在"));

        if (!article.getAuthorId().equals(userId)) {
            throw new IllegalArgumentException("无权编辑他人文章");
        }

        article.setTitle(request.getTitle());
        article.setSummary(request.getSummary());
        article.setContent(request.getContent());
        article.setCategoryId(request.getCategoryId());
        article.setCategoryName(request.getCategoryName());
        article.setTags(request.getTags());
        article.setCoverImage(request.getCoverImage());
        if (request.getStatus() != null) {
            article.setStatus(request.getStatus());
        }

        return articleRepository.save(article);
    }

    /**
     * 删除文章并同时删除其所有评论（使用事务保证一致性）。
     * 需要 MongoDB 副本集环境支持。
     */
    @Transactional
    public void delete(String id, String userId) {
        Article article = articleRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("文章不存在"));

        if (!article.getAuthorId().equals(userId)) {
            throw new IllegalArgumentException("无权删除他人文章");
        }

        commentRepository.deleteByArticleId(id);
        articleRepository.deleteById(id);
    }

    /**
     * 查看文章详情，同时对浏览量原子+1。
     * 使用 MongoTemplate 的 findAndModify 实现原子操作。
     */
    public Article getDetail(String id) {
        Query query = new Query(Criteria.where("_id").is(id));
        Update update = new Update().inc("view_count", 1);
        Article article = mongoTemplate.findAndModify(query, update,
                new org.springframework.data.mongodb.core.FindAndModifyOptions().returnNew(true),
                Article.class);

        if (article == null) {
            throw new IllegalArgumentException("文章不存在");
        }
        return article;
    }

    /**
     * 多条件筛选文章列表（分页）。
     * 使用 MongoTemplate + Criteria 动态构建查询条件。
     */
    public Page<Article> queryArticles(ArticleQuery queryParam) {
        List<Criteria> criteriaList = new ArrayList<>();

        if (StringUtils.hasText(queryParam.getCategoryId())) {
            criteriaList.add(Criteria.where("category_id").is(queryParam.getCategoryId()));
        }
        if (StringUtils.hasText(queryParam.getStatus())) {
            criteriaList.add(Criteria.where("status").is(queryParam.getStatus()));
        }
        if (queryParam.getTags() != null && !queryParam.getTags().isEmpty()) {
            criteriaList.add(Criteria.where("tags").all(queryParam.getTags()));
        }
        if (StringUtils.hasText(queryParam.getKeyword())) {
            Criteria keywordCriteria = new Criteria().orOperator(
                    Criteria.where("title").regex(queryParam.getKeyword(), "i"),
                    Criteria.where("summary").regex(queryParam.getKeyword(), "i")
            );
            criteriaList.add(keywordCriteria);
        }
        if (queryParam.getStartTime() != null) {
            criteriaList.add(Criteria.where("created_at").gte(queryParam.getStartTime()));
        }
        if (queryParam.getEndTime() != null) {
            criteriaList.add(Criteria.where("created_at").lte(queryParam.getEndTime()));
        }

        Query query = new Query();
        if (!criteriaList.isEmpty()) {
            query.addCriteria(new Criteria().andOperator(criteriaList.toArray(new Criteria[0])));
        }

        long total = mongoTemplate.count(query, Article.class);

        Sort.Direction direction = "asc".equalsIgnoreCase(queryParam.getSortDir())
                ? Sort.Direction.ASC : Sort.Direction.DESC;
        Pageable pageable = PageRequest.of(
                queryParam.getPage() - 1,
                queryParam.getSize(),
                Sort.by(direction, queryParam.getSortBy()));
        query.with(pageable);

        // 列表页不返回全文内容，排除 content 字段以提升性能
        query.fields().exclude("content");

        List<Article> articles = mongoTemplate.find(query, Article.class);
        return new PageImpl<>(articles, pageable, total);
    }
}

package com.cms.service;

import com.cms.dto.CategoryRequest;
import com.cms.entity.Category;
import com.cms.repository.CategoryRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CategoryService {

    private final CategoryRepository categoryRepository;

    public Category create(CategoryRequest request) {
        if (categoryRepository.existsByName(request.getName())) {
            throw new IllegalArgumentException("分类名称已存在");
        }
        if (categoryRepository.existsBySlug(request.getSlug())) {
            throw new IllegalArgumentException("分类别名已存在");
        }

        Category category = new Category();
        category.setName(request.getName());
        category.setSlug(request.getSlug());
        category.setParentId(request.getParentId());
        category.setSortOrder(request.getSortOrder() != null ? request.getSortOrder() : 0);
        category.setArticleCount(0);

        return categoryRepository.save(category);
    }

    /**
     * 获取完整的分类树结构。
     * 先查出所有顶级分类（parentId 为 null），再递归填充子分类。
     */
    public List<Category> getCategoryTree() {
        List<Category> roots = categoryRepository.findByParentIdIsNullOrderBySortOrderAsc();
        roots.forEach(this::fillChildren);
        return roots;
    }

    private void fillChildren(Category parent) {
        List<Category> children = categoryRepository.findByParentIdOrderBySortOrderAsc(parent.getId());
        parent.setChildren(children);
        children.forEach(this::fillChildren);
    }

    public Category update(String id, CategoryRequest request) {
        Category category = categoryRepository.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("分类不存在"));

        category.setName(request.getName());
        category.setSlug(request.getSlug());
        category.setParentId(request.getParentId());
        if (request.getSortOrder() != null) {
            category.setSortOrder(request.getSortOrder());
        }

        return categoryRepository.save(category);
    }

    public void delete(String id) {
        List<Category> children = categoryRepository.findByParentIdOrderBySortOrderAsc(id);
        if (!children.isEmpty()) {

if (StringUtils.hasText(queryParam.getCategoryId())) {
    criteriaList.add(Criteria.where("category_id").is(queryParam.getCategoryId()));

// 2. 数组包含条件：文章必须同时包含所有指定标签
if (queryParam.getTags() != null && !queryParam.getTags().isEmpty()) {
    criteriaList.add(Criteria.where("tags").all(queryParam.getTags()));
}

// 3. 模糊匹配条件：标题或摘要包含关键词（不区分大小写）
if (StringUtils.hasText(queryParam.getKeyword())) {
    Criteria keywordCriteria = new Criteria().orOperator(
            Criteria.where("title").regex(queryParam.getKeyword(), "i"),
            Criteria.where("summary").regex(queryParam.getKeyword(), "i")
    );
    criteriaList.add(keywordCriteria);
}

// 4. 时间范围条件
if (queryParam.getStartTime() != null) {
    criteriaList.add(Criteria.where("created_at").gte(queryParam.getStartTime()));
}

// 5. 将所有条件用 AND 组合
Query query = new Query();
if (!criteriaList.isEmpty()) {
    query.addCriteria(new Criteria().andOperator(criteriaList.toArray(new Criteria[0])));

package com.cms.service;

import lombok.RequiredArgsConstructor;
import org.bson.Document;
import org.springframework.data.domain.Sort;
import org.springframework.data.mongodb.core.MongoTemplate;
import org.springframework.data.mongodb.core.aggregation.*;
import org.springframework.data.mongodb.core.query.Criteria;
import org.springframework.stereotype.Service;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Map;

@Service
@RequiredArgsConstructor
public class StatisticsService {

    private final MongoTemplate mongoTemplate;

    /**
     * 每日发文量统计。
     * 按创建日期分组统计最近 30 天每天的发文数量。
     *
     * 等效的 MongoDB Shell 聚合：
     * db.articles.aggregate([
     *   { $match: { status: "PUBLISHED", created_at: { $gte: ISODate("...") } } },
     *   { $group: {
     *       _id: { $dateToString: { format: "%Y-%m-%d", date: "$created_at" } },
     *       count: { $sum: 1 }
     *   }},
     *   { $sort: { _id: 1 } }
     * ])
     */
    public List<Document> getDailyArticleCount(int days) {
        LocalDateTime startDate = LocalDateTime.now().minusDays(days);

        Aggregation aggregation = Aggregation.newAggregation(
                Aggregation.match(
                        Criteria.where("status").is("PUBLISHED")
                                .and("created_at").gte(startDate)
                ),
                Aggregation.project()
                        .and(DateOperators.DateToString.dateOf("created_at")
                                .toString("%Y-%m-%d"))
                        .as("date")
                        .and("_id").as("articleId"),
                Aggregation.group("date").count().as("count"),
                Aggregation.sort(Sort.Direction.ASC, "_id"),
                Aggregation.project()
                        .and("_id").as("date")
                        .and("count").as("count")
                        .andExclude("_id")
        );

        AggregationResults<Document> results =
                mongoTemplate.aggregate(aggregation, "articles", Document.class);
        return results.getMappedResults();
    }

    /**
     * 热门标签 Top10。
     * 将文章的 tags 数组展开后分组统计。
     *
     * 等效的 MongoDB Shell 聚合：
     * db.articles.aggregate([
     *   { $match: { status: "PUBLISHED" } },
     *   { $unwind: "$tags" },
     *   { $group: { _id: "$tags", count: { $sum: 1 } } },
     *   { $sort: { count: -1 } },
     *   { $limit: 10 },
     *   { $project: { _id: 0, tag: "$_id", count: 1 } }
     * ])
     */
    public List<Document> getHotTags(int limit) {
        Aggregation aggregation = Aggregation.newAggregation(
                Aggregation.match(Criteria.where("status").is("PUBLISHED")),
                Aggregation.unwind("tags"),
                Aggregation.group("tags").count().as("count"),
                Aggregation.sort(Sort.Direction.DESC, "count"),
                Aggregation.limit(limit),
                Aggregation.project()
                        .and("_id").as("tag")
                        .and("count").as("count")
                        .andExclude("_id")
        );

        AggregationResults<Document> results =
                mongoTemplate.aggregate(aggregation, "articles", Document.class);
        return results.getMappedResults();
    }

    /**
     * 用户活跃度排行——按文章数和评论数综合排序。
     * 使用 $lookup 关联 users、articles、comments 三个集合。
     *
     * 等效的 MongoDB Shell 聚合：
     * db.users.aggregate([
     *   { $lookup: {
     *       from: "articles",
     *       localField: "_id",
     *       foreignField: "author_id",
     *       as: "user_articles"
     *   }},
     *   { $lookup: {
     *       from: "comments",
     *       localField: "_id",
     *       foreignField: "user_id",
     *       as: "user_comments"
     *   }},
     *   { $project: {
     *       nickname: 1,
     *       username: 1,
     *       article_count: { $size: "$user_articles" },
     *       comment_count: { $size: "$user_comments" },
     *       activity_score: { $add: [
     *         { $multiply: [{ $size: "$user_articles" }, 5] },
     *         { $size: "$user_comments" }
     *       ]}
     *   }},
     *   { $sort: { activity_score: -1 } },
     *   { $limit: 10 }
     * ])
     */
    public List<Document> getActiveUsers(int limit) {
        Aggregation aggregation = Aggregation.newAggregation(
                Aggregation.lookup("articles", "_id", "author_id", "user_articles"),
                Aggregation.lookup("comments", "_id", "user_id", "user_comments"),
                Aggregation.project("nickname", "username")
                        .and(ArrayOperators.Size.lengthOfArray(
                                ConditionalOperators.ifNull("user_articles").then(List.of())))
                        .as("articleCount")
                        .and(ArrayOperators.Size.lengthOfArray(
                                ConditionalOperators.ifNull("user_comments").then(List.of())))
                        .as("commentCount"),
                Aggregation.addFields()
                        .addFieldWithValue("activityScore",
                                ArithmeticOperators.Add.valueOf(
                                        ArithmeticOperators.Multiply.valueOf("articleCount").multiplyBy(5)
                                ).add("commentCount"))
                        .build(),
                Aggregation.sort(Sort.Direction.DESC, "activityScore"),
                Aggregation.limit(limit),
                Aggregation.project("nickname", "username", "articleCount", "commentCount", "activityScore")
        );

        AggregationResults<Document> results =
                mongoTemplate.aggregate(aggregation, "users", Document.class);
        return results.getMappedResults();
    }

14.6.1 Change Streams 实现新评论实时通知

当有用户在文章下发表新评论时，我们希望实时推送通知给正在浏览该文章的用户。实现方案：Change Streams 监听 comments 集合的插入事件 → 通过 SSE（Server-Sent Events）推送给前端。

监听器 CommentChangeStreamListener：

package com.cms.listener;

import lombok.extern.slf4j.Slf4j;
import org.bson.Document;
import org.springframework.data.mongodb.core.messaging.Message;
import org.springframework.data.mongodb.core.messaging.MessageListener;
import org.springframework.stereotype.Component;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;

import java.io.IOException;
import java.util.List;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.CopyOnWriteArrayList;

@Slf4j
@Component
public class CommentChangeStreamListener implements MessageListener<Document, Document> {

    /**
     * 存储每篇文章的 SSE 连接列表。
     * key = articleId, value = 订阅该文章的 SSE 连接列表。
     */
    private final Map<String, List<SseEmitter>> articleEmitters = new ConcurrentHashMap<>();

    /**
     * 为某篇文章注册一个 SSE 连接。
     */
    public SseEmitter subscribe(String articleId) {
        SseEmitter emitter = new SseEmitter(300_000L); // 5分钟超时
        articleEmitters.computeIfAbsent(articleId, k -> new CopyOnWriteArrayList<>()).add(emitter);

        emitter.onCompletion(() -> removeEmitter(articleId, emitter));
        emitter.onTimeout(() -> removeEmitter(articleId, emitter));
        emitter.onError(e -> removeEmitter(articleId, emitter));

        return emitter;
    }

    private void removeEmitter(String articleId, SseEmitter emitter) {
        List<SseEmitter> emitters = articleEmitters.get(articleId);
        if (emitters != null) {
            emitters.remove(emitter);
            if (emitters.isEmpty()) {
                articleEmitters.remove(articleId);
            }
        }
    }

    /**
     * 收到 Change Stream 事件时触发。
     * 解析新评论数据，推送给订阅对应文章的所有 SSE 客户端。
     */
    @Override
    public void onMessage(Message<Document, Document> message) {
        Document fullDocument = message.getBody();
        if (fullDocument == null) {
            return;
        }

        String articleId = fullDocument.getString("article_id");
        String userNickname = fullDocument.getString("user_nickname");
        String content = fullDocument.getString("content");

        log.info("收到新评论 - 文章:{}, 用户:{}, 内容:{}", articleId, userNickname, content);

        List<SseEmitter> emitters = articleEmitters.get(articleId);
        if (emitters == null || emitters.isEmpty()) {
            return;
        }

        String jsonPayload = """
                {"articleId":"%s","userNickname":"%s","content":"%s"}
                """.formatted(articleId, userNickname, content).trim();

        for (SseEmitter emitter : emitters) {
            try {
                emitter.send(SseEmitter.event()
                        .name("new-comment")
                        .data(jsonPayload));
            } catch (IOException e) {
                removeEmitter(articleId, emitter);
            }
        }
    }
}

    }
}

14.6.2 事务示例：删除文章同时清理评论

事务的完整实现已在 ArticleService.delete() 方法中给出（见 14.4.4 节）。这里做详细说明和补充一个更完整的示例：

package com.cms.service;

import com.cms.entity.Article;
import com.cms.repository.ArticleRepository;
import com.cms.repository.CommentRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.mongodb.core.MongoTemplate;
import org.springframework.data.mongodb.core.query.Criteria;
import org.springframework.data.mongodb.core.query.Query;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Slf4j
@Service
@RequiredArgsConstructor
public class ArticleTransactionDemo {

    private final ArticleRepository articleRepository;
    private final CommentRepository commentRepository;
    private final MongoTemplate mongoTemplate;

    /**
     * 事务性删除文章及其所有关联数据。
     *
     * @Transactional 注解确保以下操作要么全部成功，要么全部回滚：
     * 1. 删除文章下的所有评论
     * 2. 删除 GridFS 中的封面图片记录（可选）
     * 3. 删除文章本身
     *
     * 注意：MongoDB 事务需要副本集环境。
     */
    @Transactional
    public void deleteArticleWithCleanup(String articleId, String userId) {
        Article article = articleRepository.findById(articleId)
                .orElseThrow(() -> new IllegalArgumentException("文章不存在"));

        if (!article.getAuthorId().equals(userId)) {
            throw new IllegalArgumentException("无权删除他人文章");
        }

        // 步骤1：删除该文章下的所有评论
        long deletedComments = mongoTemplate.remove(
                new Query(Criteria.where("article_id").is(articleId)),
                "comments"
        ).getDeletedCount();
        log.info("删除文章[{}]的评论 {} 条", articleId, deletedComments);

        // 步骤2：删除文章文档
        articleRepository.deleteById(articleId);
        log.info("文章[{}]已删除", articleId);

        // 如果在步骤 1 和步骤 2 之间发生异常，事务将回滚，
        // 已删除的评论也会被恢复，保证数据一致性。
    }
}

事务的关键要点：

  1. MongoDB 事务要求 副本集（Replica Set） 或分片集群环境
  2. 需要配置 MongoTransactionManager Bean（见 14.3.7 节）
  3. 使用 Spring 的 @Transactional 注解即可声明式管理事务
  4. 事务内的操作受到 60 秒超时限制（默认）
  5. 事务内不能执行创建集合或创建索引等 DDL 操作

14.6.3 TTL 索引：自动清理过期验证码

TTL（Time-To-Live）索引可以让 MongoDB 自动删除过期文档，非常适合验证码、临时 Token 等场景。

实体类（已在 14.2.5 节定义）：

@Data
@Document(collection = "verification_codes")
public class VerificationCode {
    @Id
    private String id;

    @Indexed(unique = true)
    private String email;

    private String code;

    @Indexed(expireAfterSeconds = 300)  // 5分钟后自动删除
    @Field("created_at")
    private Instant createdAt;
}

Repository：

package com.cms.repository;

import com.cms.entity.VerificationCode;
import org.springframework.data.mongodb.repository.MongoRepository;

import java.util.Optional;

public interface VerificationCodeRepository extends MongoRepository<VerificationCode, String> {

    Optional<VerificationCode> findByEmailAndCode(String email, String code);

    void deleteByEmail(String email);
}

Service 实现：

package com.cms.service;

import com.cms.entity.VerificationCode;
import com.cms.repository.VerificationCodeRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.time.Instant;
import java.util.Random;

@Slf4j
@Service
@RequiredArgsConstructor
public class VerificationCodeService {

    private final VerificationCodeRepository codeRepository;

    /**
     * 生成6位数字验证码并存储。
     * TTL 索引会在 300 秒（5分钟）后自动删除该文档。
     */
    public String generateCode(String email) {
        // 先删除该邮箱的旧验证码
        codeRepository.deleteByEmail(email);

        String code = String.format("%06d", new Random().nextInt(999999));

        VerificationCode verificationCode = new VerificationCode();
        verificationCode.setEmail(email);
        verificationCode.setCode(code);
        verificationCode.setCreatedAt(Instant.now());

        codeRepository.save(verificationCode);
        log.info("已生成验证码 [{}] -> {}, 5分钟后自动过期", email, code);

        return code;
    }

    /**
     * 验证码校验。
     * 如果文档已被 TTL 索引自动删除，则查询不到，校验自然失败。
     */
    public boolean verifyCode(String email, String code) {
        return codeRepository.findByEmailAndCode(email, code).isPresent();
    }
}

工作原理说明：

  • @Indexed(expireAfterSeconds = 300) 在 created_at 字段上创建 TTL 索引
  • MongoDB 后台线程每 60 秒扫描一次，删除 created_at + 300秒 < 当前时间 的文档
  • 实际删除可能有最多 60 秒的延迟（取决于后台扫描间隔）
  • TTL 索引只能建在类型为 Date / Instant 的字段上
